-- V-GUI: Rất nhẹ và ít mã hóa.
local vgui = loadstring(game:HttpGet("https://raw.githubusercontent.com/Green-02/V-GUI-Library/main/Source"))()
local Window = vgui:Window("Cận đẹp zai - HUB KIỂM TRA", UDim2.fromOffset(400, 300))

-- Tạo nút ẩn/hiện GUI (Icon Bánh răng ĐEN)
local ImageButton = Instance.new("ImageButton")
ImageButton.Name = "ToggleImageButton"
ImageButton.Parent = game.CoreGui
ImageButton.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
ImageButton.BorderSizePixel = 0
ImageButton.Position = UDim2.new(0.10615778, 0, 0.16217947, 0)
ImageButton.Size = UDim2.new(0, 40, 0, 40)
ImageButton.Draggable = true
ImageButton.Image = "http://www.roblox.com/asset/?id=6207038140"

local isHidden = false
ImageButton.MouseButton1Down:Connect(function()
    isHidden = not isHidden
    Window:SetVisible(not isHidden)
end)

-- Tab 1: Thông Tin
local TabInfo = Window:Tab("Thông Tin")
TabInfo:Button("Discord", "TG Community", function()
    setclipboard("https://discord.gg/MxSpY55t")
end)

-- Tab 2: Blox Fruit (Chỉ 1 script)
local TabBF = Window:Tab("Blox Fruit")
TabBF:Button("Hiru hub", "Script Blox Fruit", function()
    getgenv().Settings = {
        JoinTeam = true,
        Team = "Marines"
    }
    loadstring(game:HttpGet("https://raw.githubusercontent.com/kiddohiru/Source/main/BloxFruits.lua"))()
end)

-- Hiển thị cửa sổ
Window:SetVisible(true)
