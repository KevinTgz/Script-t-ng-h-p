-- Đây là cách duy nhất để chạy tất cả các script mà Krnl không chặn

local CoreGui = game:GetService("CoreGui")

-- 1. Tạo GUI Tối thiểu để báo hiệu hoạt động
local Notification = Instance.new("ScreenGui")
Notification.Name = "TG_Hub_Status"
Notification.Parent = CoreGui
Notification.ZIndexBehavior = Enum.ZIndexBehavior.Sibling

local Label = Instance.new("TextLabel")
Label.Parent = Notification
Label.Size = UDim2.new(0, 300, 0, 80)
Label.Position = UDim2.new(0.5, -150, 0.5, -40)
Label.Text = "TG Hub Đang Khởi Động..."
Label.BackgroundColor3 = Color3.fromRGB(50, 50, 50)
Label.TextColor3 = Color3.fromRGB(255, 255, 255)
Label.BorderSizePixel = 0
Label.Font = Enum.Font.SourceSansBold
Label.TextWrapped = true

-- 2. Hàm Thực thi Script (Hàm Chính)
local function execute_loader(name, url)
    Label.Text = "Đang tải: " .. name .. "..."
    local success, script_loader = pcall(game.HttpGet, game, url)
    
    if success and script_loader then
        Label.Text = "Thực thi: " .. name .. "..."
        local exec_success, exec_result = pcall(loadstring(script_loader))
        if exec_success then
            Label.Text = "Đã thực thi " .. name .. " thành công!"
            wait(2)
        else
            Label.Text = "LỖI THỰC THI: " .. name .. "\n" .. tostring(exec_result)
            wait(4)
        end
    else
        Label.Text = "LỖI TẢI: " .. name .. "\nKiểm tra lại URL."
        wait(4)
    end
end

-- 3. Danh sách tất cả các script

    -- **99 Nights in Forest (Đã thêm lại)**
    { Name = "Foxname (99NIF)", URL = "https://raw.githubusercontent.com/caomod2077/Script/refs/heads/main/FoxnameHub.lua" },

        setclipboard("https://discord.gg/MxSpY55t")
        Label.Text = "Đã copy Discord link và Tiktok link!"

    end },
    { Name = "Steal/Troll Info", URL = nil, Action = function() 
        Label.Text = "Script Steal Brainrot và Troll sẽ được cập nhật sớm!"
        wait(2)
    end },
}

-- 4. Chạy Từng Script (Bạn sẽ cần sửa lại logic này nếu không muốn chạy tự động)
Label.Text = "Sẵn sàng chạy từng script!"
wait(2)

for i, script_data in ipairs(ScriptList) do
    if script_data.URL then
        execute_loader(script_data.Name, script_data.URL)
    elseif script_data.Action then
        script_data.Action()
    end
end


-- 5. Xóa thông báo sau khi hoàn tất
Label.Text = "Tất cả các script đã được thực thi!"
